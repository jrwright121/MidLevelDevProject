/**
 * @fileoverview Function to convert number digits to all possible combinations of letters as they correspond on a phone key pad.
 * @author Jacob Wright
 * @param {input} - any length number segment
 * @returns {string[]} - all letter combinations
 */
exports.getAllVanityWords = function (input) {
    console.log('Entered getAllVanityWords: ', input);
    var number = input.split('');
    var hashTable = {
        '0': ['0'],
        '1': ['1'],
        "2": ["a", "b", "c"],
        "3": ["d", "e", "f"],
        "4": ["g", "h", "i"],
        "5": ["j", "k", "l"],
        "6": ["m", "n", "o"],
        "7": ["p", "q", "r", "s"],
        "8": ["t", "u", "v"],
        "9": ["w", "x", "y", "z"]
    };
    var ans = [];
    // Recursive function to build array of all combinations of letter mapped to hashedTable
    var backtrack = function (combo, digits) {
        if (digits.length === 0) {
            ans.push(combo);
        }
        else {
            // @ts-ignore
            hashTable[digits[0]].forEach(function (letter) {
                // calls itself iterating on all last combination and adding all additional combinations
                // until all digits have been processed. 
                backtrack("" + combo + letter, digits.slice(1));
            });
        }
    };
    backtrack('', number);
    return ans;
};
